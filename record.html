git

    官网下载安装包，直接安装即可 设置用户签名
    git config --global user.name aa
    git config --global user.email aa
    git init 初始化本地库
    git status 查看状态
    git add ./hellow.txt 添加暂存区
    git rm --cached hello.txt 删除暂存区文件
    git commit -m “123” hello.txt 提交到本地仓库
    git reflog 查看版本信息
    git log 查看版本详细信息
    git reset --hard 5770506 切换版本，穿越版本（查看其它版本)
    分支
    git branch -v 查看分支
    git branch hot-fix 创建分支
    git checkout hot-fix 切换分支
    git merge hot-fix 合并分支,站在master合并hot-fix
